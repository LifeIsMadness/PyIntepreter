a1 = []
a2 = []
n = int(input())
for i in range(n):
	a1 = a1 + []
	for j in range(n):
		a1[i] = a1[i] + int(input())

for i in range(n):
	a2 = a2 + []
	for j in range(n):
		a2[i] = a2[i] + int(input())
res = []
for i in range(n):
	res = res + []
	for j in range(n):
		res[i] = res[i] + 0
for i in range(n):
	for j in range(n):
		for k in range(n):
			res[i][j] =  res[i][j] + a1[i][k] * a2[k][j]
print(res)
